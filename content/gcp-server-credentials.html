SQL Query to Generate Truncate SQL Commands for All Tables in a Schema


SELECT 'TRUNCATE TABLE ' || table_schema || '.' || table_name || ' CASCADE;' AS truncate_statement
FROM information_schema.tables
WHERE table_schema = 'mdms';
